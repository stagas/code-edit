<code-edit />

installation:

npm i code-edit

Also, in order to use ready made syntax highlighting from code-syntax and themes from plenty-themes. This is optional as you can define your own syntax and theme as well using the same api as in code-syntax.

npm i code-syntax
npm i plenty-themes # a collection of themes

<code-edit autofocus autoresize language="js" theme="deep"> ... </code-edit>

class Vector {
  x: number
  y: number
  constructor(x: number, y: number) {
    this.x = x
    this.y = y
  }
  add(v: Vector): Vector {
    return new Vector(this.x + v.x, this.y + v.y)
  }
}

how to use:

<script type="module">
  // register the custom element
  import { CodeEditElement, languages } from 'code-edit'
  customElements.define('code-edit', CodeEditElement)

  // using a code-syntax highlight:
  import 'plenty-themes/deep.css'
  import 'code-syntax/themes/default.css'
  languages.js = import('code-syntax/languages/js') // no await, promise is fine
</script>

<!-- the <pre> tag is optional but makes editors retain formatting -->
<pre><code-edit language="js" theme="deep">
